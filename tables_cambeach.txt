User table:
	id primary key,				
	name varchar,				"Name Complete"
	age int,				18
	email varchar,				"email@example.com"
	password varchar,			hash
	telephone varchar(20),			"11 99999-9999"
	cpf varchar(14),			123.456.789-00
	category foreing key category,		object category
	created_at date				2025-12-25

Tournament table:
	id primary key,
	name varchar,				"Name complete"
	local varchar,				"Local of tournament"
	organization varchar,			"Name of organization"
	start_date date,			2025-12-05
	end_date date,				2025-12-09
	n_players small_int, 			32

	tournament foreing key tournament,	object tournament
	team_1 foreing key team,		object team

	match_state varchar,			"Agendado" ...
	court small_int,			4
	games_team_1 small_int,			4
	games_team_2 small_int,			6	
	start_time datetime,			2025-12-06 HH:mm:ss
	end_time datetime,			2025-12-06 HH:mm:ss
